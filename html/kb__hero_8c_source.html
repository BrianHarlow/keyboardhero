178  else if(setColumn == 2){
179  xmin = 22;
180  xmax = 29;
181  Terminal_SetColor(PLAYER1_UART, BackgroundYellow);
182  }
183  else if(setColumn == 3){
184  xmin = 32;
185  xmax = 39;
186  Terminal_SetColor(PLAYER1_UART, BackgroundCyan);
187  }
188  else if(setColumn == 4){
189  xmin = 42;
190  xmax = 49;
191  Terminal_SetColor(PLAYER1_UART, BackgroundMagenta);
192  }
193  else{
194  xmin = 2;
195  xmax = 9;
196  Terminal_SetColor(PLAYER1_UART, BackgroundGreen);
197  }
198  char ymin;
199  char ymax;
200  volatile uint8_t i;
201  note_t * note = 0;
202  // Find an unused note object
203  for(i = 0; i < KBH_MAX_NOTES; i++) if(notes[i].status == 0) note = &notes[i];
204  if(note) {
205  note->status = 1;
206  note->xmin = xmin;
207  note->xmax = xmax;
208  note->ymin = 6;
209  note->ymax = 9;
210  // use the task scheduler to move the char down and right at the calculated
211  // period
212  Task_Schedule((task_t)MoveDownNote, note, KBH_NOTE_SPEED, KBH_NOTE_SPEED);
213  Game_FillRect(' ', note->xmin, note->ymin, note->xmax, note->ymax);
214  Terminal_SetColor(PLAYER1_UART, BackgroundBlack);
215  }
216  Task_Schedule(SendNote, 0, random_int(MIN_NOTE_RATE, MAX_NOTE_RATE), 0);
217 }
